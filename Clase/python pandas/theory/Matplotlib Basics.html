import numpy as np
import matplotlib.pyplot as plt

ax = plt.subplot(1,1,1)

t = np.arange(0.0, 5.0, 0.01)
s = np.cos(2*np.pi*t)
line, = ax.plot(t, s, lw=2)

ax.annotate('local max', xy=(2, 1), xytext=(3, 1.5),
            arrowprops=dict(facecolor='black', shrink=0.1),
            )

plt.ylim(-2,2)
plt.show()
